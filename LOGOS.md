fn cosine_sim_simd(
    a: SIMD[DType.float32, 1024],
    b: SIMD[DType.float32, 1024]
) -> Float32:
    # SIMD dot product + norms
    let dot = (a * b).reduce_add()
    let norm_a = (a * a).reduce_add().sqrt()
    let norm_b = (b * b).reduce_add().sqrt()
